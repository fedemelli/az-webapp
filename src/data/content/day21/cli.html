Comandi CLI Essenziali

# Creare una Public DNS Zone
az network dns zone create \\
  --resource-group MyRG \\
  --name contoso.com

# Creare un Record A
az network dns record-set a add-record \\
  --resource-group MyRG \\
  --zone-name contoso.com \\
  --record-set-name www \\
  --ipv4-address 20.45.12.34

# Creare una Private DNS Zone
az network private-dns zone create \\
  --resource-group MyRG \\
  --name app.internal

# Linkare una VNet con Auto-registration
az network private-dns link vnet create \\
  --resource-group MyRG \\
  --zone-name app.internal \\
  --name MyLink \\
  --virtual-network MyVNet \\
  --registration-enabled true
