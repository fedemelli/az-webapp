Comandi CLI Essenziali

# Creare un Action Group (Email)
az monitor action-group create \\
  --resource-group MyRG \\
  --name MyActionGroup \\
  --action email admin admin@contoso.com

# Creare un Metric Alert (CPU > 80%)
az monitor metrics alert create \\
  --name HighCPUAlert \\
  --resource-group MyRG \\
  --scopes MyVMID \\
  --condition "avg Percentage CPU > 80" \\
  --window-size 5m \\
  --evaluation-frequency 1m \\
  --action MyActionGroup
