# Creare un App Gateway
az network application-gateway create \\
  --name MyAppGateway \\
  --resource-group MyRG \\
  --location eastus \\
  --sku WAF_v2 \\
  --capacity 2 \\
  --vnet-name MyVNet \\
  --subnet MySubnet \\
  --public-ip-address MyPublicIP \\
  --http-settings-cookie-based-affinity Enabled
