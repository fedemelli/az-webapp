Comandi CLI Essenziali

# Creare un Azure Container Registry
az acr create \\
  --name MyRegistry123 \\
  --resource-group MyRG \\
  --sku Basic \\
  --admin-enabled true

# Costruire un'immagine con ACR Tasks (senza Docker locale)
az acr build --registry MyRegistry123 --image myapp:v1 .

# Eseguire un container in ACI
az container create \\
  --resource-group MyRG \\
  --name mycontainer \\
  --image MyRegistry123.azurecr.io/myapp:v1 \\
  --dns-name-label myapp-demo \\
  --ports 80
