Comandi CLI

# Creare un Availability Set
az vm availability-set create \\
  --name MyAvSet \\
  --resource-group MyRG \\
  --platform-fault-domain-count 2 \\
  --platform-update-domain-count 5

# Creare una VM nell'Availability Set
az vm create \\
  --name MyVM \\
  --resource-group MyRG \\
  --image Ubuntu2204 \\
  --availability-set MyAvSet
