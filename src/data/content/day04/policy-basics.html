Struttura di una Policy Definition

{
  "properties": {
    "displayName": "Allowed locations",
    "description": "Restrict resource deployment to specific Azure regions",
    "mode": "Indexed",
    "metadata": {
      "version": "1.0.0",
      "category": "General"
    },
    "parameters": {
      "allowedLocations": {
        "type": "Array",
        "metadata": {
          "displayName": "Allowed locations",
          "strongType": "location"
        }
      }
    },
    "policyRule": {
      "if": {
        "not": {
          "field": "location",
          "in": "[parameters('allowedLocations')]"
        }
      },
      "then": {
        "effect": "deny"
      }
    }
  }
}

Gestire Policy con Azure CLI

# Visualizzare policy definitions built-in
az policy definition list \
  --query "[?policyType=='BuiltIn']" \
  --output table

# Creare una custom policy definition
az policy definition create \
  --name "require-tag-department" \
  --display-name "Require Department Tag" \
  --description "Requires a Department tag on resources" \
  --rules policy-rule.json \
  --params policy-params.json \
  --mode Indexed

# Creare un assignment
az policy assignment create \
  --name "allowed-locations-assignment" \
  --display-name "Allowed Locations - West Europe" \
  --scope "/subscriptions/{subscription-id}" \
  --policy "/providers/Microsoft.Authorization/policyDefinitions/e56962a6..." \
  --params '{"allowedLocations": {"value": ["westeurope", "northeurope"]}}'

# Verificare compliance
az policy state list \
  --resource-group "my-rg" \
  --query "[?complianceState=='NonCompliant']"

Gestire Policy con PowerShell

# Get built-in policy definitions
Get-AzPolicyDefinition | Where-Object { $_.Properties.PolicyType -eq "BuiltIn" }

# Get specific policy definition
$definition = Get-AzPolicyDefinition -Name "e56962a6-4747-49cd-b67b-bf8b01975c4c"

# Create policy assignment
$params = @{
  "allowedLocations" = @("westeurope", "northeurope")
}

New-AzPolicyAssignment `
  -Name "allowed-locations" `
  -DisplayName "Allowed Locations" `
  -Scope "/subscriptions/{subscription-id}" `
  -PolicyDefinition $definition `
  -PolicyParameterObject $params

# Get non-compliant resources
Get-AzPolicyState -Filter "complianceState eq 'NonCompliant'"
