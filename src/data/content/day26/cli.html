Comandi CLI Essenziali

# Creare un Recovery Services Vault
az backup vault create \\
  --resource-group MyRG \\
  --name MyVault \\
  --location eastus

# Creare una Policy di Backup
az backup policy create \\
  --resource-group MyRG \\
  --vault-name MyVault \\
  --name MyPolicy \\
  --backup-management-type AzureIaasVM \\
  --policy policy.json

# Abilitare il backup per una VM
az backup protection enable-for-vm \\
  --resource-group MyRG \\
  --vault-name MyVault \\
  --vm MyVM \\
  --policy-name MyPolicy
