Comandi CLI Essenziali

# Creare un NSG
az network nsg create \\
  --resource-group MyRG \\
  --name MyNSG

# Creare una regola (Allow HTTP)
az network nsg rule create \\
  --resource-group MyRG \\
  --nsg-name MyNSG \\
  --name AllowHTTP \\
  --priority 100 \\
  --destination-port-ranges 80 \\
  --access Allow \\
  --protocol Tcp

# Associare NSG a una Subnet
az network vnet subnet update \\
  --resource-group MyRG \\
  --vnet-name MyVNet \\
  --name MySubnet \\
  --network-security-group MyNSG
