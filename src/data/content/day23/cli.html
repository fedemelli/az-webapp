Comandi CLI Essenziali

# Creare un Public IP Standard
az network public-ip create \\
  --resource-group MyRG \\
  --name MyPublicIP \\
  --sku Standard

# Creare il Load Balancer
az network lb create \\
  --resource-group MyRG \\
  --name MyLB \\
  --sku Standard \\
  --public-ip-address MyPublicIP \\
  --frontend-ip-name MyFrontEnd \\
  --backend-pool-name MyBackEndPool

# Creare Health Probe
az network lb probe create \\
  --resource-group MyRG \\
  --lb-name MyLB \\
  --name MyProbe \\
  --protocol tcp \\
  --port 80

# Creare Load Balancing Rule
az network lb rule create \\
  --resource-group MyRG \\
  --lb-name MyLB \\
  --name MyRule \\
  --protocol tcp \\
  --frontend-port 80 \\
  --backend-port 80 \\
  --frontend-ip-name MyFrontEnd \\
  --backend-pool-name MyBackEndPool \\
  --probe-name MyProbe
