Comandi CLI per Gestire Ruoli Entra ID

# Elencare ruoli Entra ID
az rest --method GET \
  --url "https://graph.microsoft.com/v1.0/directoryRoles"

# Con PowerShell (Microsoft Graph)
Connect-MgGraph -Scopes "RoleManagement.Read.Directory"

# Ottenere tutti i ruoli disponibili
Get-MgDirectoryRole

# Elencare membri di un ruolo specifico
$roleId = (Get-MgDirectoryRole -Filter "displayName eq 'Global Administrator'").Id
Get-MgDirectoryRoleMember -DirectoryRoleId $roleId

# Assegnare un ruolo a un utente
$userId = (Get-MgUser -UserId "mario@contoso.com").Id
New-MgDirectoryRoleMemberByRef `
  -DirectoryRoleId $roleId `
  -OdataId "https://graph.microsoft.com/v1.0/users/$userId"
